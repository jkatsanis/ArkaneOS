ASM TO BIN
nasm -f bin -o Boot.bin Boot.asm

IMG FILE
dd if=Boot.bin of=Boot.img bs=512 conv=notrunc

FLP FILE

C OBJ TO BIN 
objcopy -O binary my_program.o my_program.bin

C TO OBJ
gcc -c my_program.c -o my_program.o

MERGE
cat one.bin two.bin > result.bin

QEMU EMULATOR
sudo qemu-system-x86_64 -fda result.img
